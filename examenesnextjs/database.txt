CREATE TABLE PERSON (
	NAME VARCHAR(30),
    LASTNAME VARCHAR(30),
    CODE VARCHAR(10),
    PSSWORD VARCHAR(30),
    ROLE VARCHAR(20),
	PRIMARY KEY(CODE)
	
);

CREATE TABLE EXAM (
	CODEID VARCHAR(10),
   	CODEACC VARCHAR(10),
    TITLE VARCHAR (50),
    DESCRIPTION VARCHAR (80),
	PRIMARY KEY (CODEID)
);


CREATE TABLE QUESTION (
    QUESTIONID VARCHAR(10),
    QUESTIONTEXT VARCHAR(60),
    PERCENTAGE NUMERIC(),
    OPTIONA TEXT,
    OPTIONB TEXT,
    OPTIONC TEXT,
    OPTIOND TEXT,
    CORRECTOPTION CHAR(),
    FOREIGN KEY (QUESTIONID) REFERENCES EXAM(CODEID)
);
